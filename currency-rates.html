Получение курсов валют

Курсы валют можно получить, используя сервис DisplayCurrencyLLSRQ.

В запросе необходимо указать:

  • /DisplayCurrencyRQ/CountryCode — код страны текущего PCC
  • /DisplayCurrencyRQ/CurrencyCode — код валюты с символом звездочка (*) на конце

Пример запроса

<DisplayCurrencyRQ ReturnHostCommand="true" Version="2.1.0" xmlns="http://webservices.sabre.com/sabreXML/2011/10">
  <CountryCode>RU</CountryCode>
  <CurrencyCode>EUR*</CurrencyCode>
</DisplayCurrencyRQ>

Пример ответа

<DisplayCurrencyRS Version="2.1.0" xmlns="http://webservices.sabre.com/sabreXML/2011/10" xmlns:stl="http://services.sabre.com/STL/v01" xmlns:xs="http://www.w3.org/2001/XMLSchema" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance">
  <stl:ApplicationResults status="Complete">
    <stl:Success timeStamp="2016-01-31T09:58:21-06:00">
      <stl:SystemSpecificResults>
        <stl:HostCommand LNIATA="3731E9">DC*RU/EUR*</stl:HostCommand>
      </stl:SystemSpecificResults>
    </stl:Success>
  </stl:ApplicationResults>
  <Country Name="EUROPEAN UNION" RPH="1">
    <CurrencyCode>EUR</CurrencyCode>
    <CurrencyName>EURO</CurrencyName>
    <DecimalPlaces>2</DecimalPlaces>
    <Rate Type="BSR">84.500068022554</Rate>
  </Country>
</DisplayCurrencyRS>
